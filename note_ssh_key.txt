//add new key for gitbub

ls ~/.ssh
ssh-keygen -t ed25519 -f ~/.ssh/id_ting -C "github"
cat ~/.ssh/id_ting.pub

In GitHub：
Settings → SSH and GPG keys → New SSH key
Key：paste the result of cat ~/.ssh/id_ting.pub

------------------------------------------------------
//setting two key for two account
nano ~/.ssh/config
# ===== Ting-Repo (new account) =====
Host github-ting
  HostName github.com
  User git
  IdentityFile ~/.ssh/id_ting
  IdentitiesOnly yes

# ===== Old GitHub account =====
Host github-private
  HostName github.com
  User git
  IdentityFile ~/.ssh/id_rsa
  IdentitiesOnly yes
